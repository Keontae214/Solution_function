In [9]:
def CobbDouglas(K,L,A,alpha):
    P=A*(K**alpha)*(L**(1-alpha));
    return P;

L=1; alpha=0.3;

K=np.linspace(0,5,100);

A=np.array([1,2,4,5]);
P=np.zeros((4,100));

for i in range(len(A)):
    P[i,:]=CobbDouglas(K,L,A[i],alpha);
    
for j in range(len(A)):
    plt.plot(K,P[j,:]);
    
plt.title("Cobb-Douglas Production");
plt.xlabel("K - Capital");
plt.ylabel("P - Total Production");
plt.legend(['A=1','A=2','A=4','A=5']);
plt.show()
